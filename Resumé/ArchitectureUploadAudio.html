// config/multerConfig.js
const multer = require('multer');
const path = require('path');

// Configuration du stockage
const storage = multer.diskStorage({
  destination: function(req, file, cb) {
    // D√©terminer le dossier selon le type de fichier
    if (file.fieldname === 'audio') {
      cb(null, 'uploads/audios/');
    } else if (file.fieldname === 'image') {
      cb(null, 'uploads/images/');
    }
  },
  
  filename: function(req, file, cb) {
    // Cr√©er un nom unique : timestamp-random.extension
    const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
    cb(null, uniqueSuffix + path.extname(file.originalname));
  }
});

// Validation des fichiers
const fileFilter = (req, file, cb) => {
  if (file.fieldname === 'audio') {
    const audioTypes = ['audio/mpeg', 'audio/mp3', 'audio/wav'];
    if (audioTypes.includes(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error('Format audio non support√©'), false);
    }
  } else if (file.fieldname === 'image') {
    const imageTypes = ['image/jpeg', 'image/png', 'image/jpg'];
    if (imageTypes.includes(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error('Format image non support√©'), false);
    }
  }
};

const upload = multer({
  storage: storage,
  fileFilter: fileFilter,
  limits: { fileSize: 50 * 1024 * 1024 }  // Limite 50MB
});

module.exports = upload;
